思路和坑点

　　对每个课程建立选课的学生表，然后排序输出即可。

AC代码

#include<iostream>
#include<stdlib.h>
#include<stdio.h>
#include<string.h>
#include<math.h>
#include<algorithm>
#include<vector>
#include<string>
#include<map>
#include<unordered_map>
#include<queue>
#include<set>
#include<stack>
using namespace std;

int main(void){
#ifdef ONLINE_JUDGE
#else
    freopen("1.txt","r",stdin);
#endif
    int n,k;
    scanf("%d%d",&n,&k);
    vector<string> v[k+1];
    for(int i=0;i<n;i++){
        string name;
        int m;
        cin>>name; scanf("%d",&m);      //读入数据 
        for(int j=0;j<m;j++){
            int id;
            scanf("%d",&id);
            v[id].push_back(name);      //插入到相应的课程名单中 
        }
    }
    for(int i=1;i<=k;i++){
        int size=v[i].size();
        printf("%d %d\n",i,size);
        if(size) sort(v[i].begin(),v[i].end()); //如果不空排序输出 
        for(int j=0;j<size;j++){
            printf("%s\n",v[i][j].c_str());
        }
    }
    return 0;
}
